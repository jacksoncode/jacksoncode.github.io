# Python 函数与模块

## 1. 函数定义

### 基本函数

```python
# 无参数函数
def greet():
    return "Hello, World!"

# 带参数函数
def greet(name):
    return f"Hello, {name}!"

# 带默认参数
def greet(name, greeting="Hello"):
    return f"{greeting}, {name}!"

# 多返回值
def get_stats(numbers):
    return min(numbers), max(numbers), sum(numbers) / len(numbers)

min_val, max_val, avg_val = get_stats([1, 2, 3, 4, 5])
```

### 位置参数与关键字参数

```python
def power(base, exponent=2):
    return base ** exponent

# 位置参数
power(3)      # 9
power(3, 3)   # 27

# 关键字参数
power(base=3, exponent=4)  # 81
power(exponent=4, base=3)  # 81

# 混合使用 (位置参数必须在关键字参数前)
power(3, exponent=4)  # 81
```

### *args 和 **kwargs

```python
# *args - 可变数量位置参数
def sum_all(*args):
    return sum(args)

print(sum_all(1, 2, 3, 4, 5))  # 15

# **kwargs - 可变数量关键字参数
def print_info(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

print_info(name="Alice", age=30, city="Beijing")

# 组合使用
def func(*args, **kwargs):
    print(f"args: {args}")
    print(f"kwargs: {kwargs}")

func(1, 2, 3, name="Bob", age=25)
```

## 2. Lambda 函数

```python
# 基本语法
square = lambda x: x ** 2
print(square(5))  # 25

# 带多个参数
add = lambda x, y: x + y
print(add(3, 4))  # 7

# 与内置函数结合
numbers = [1, 2, 3, 4, 5]

# sorted
students = [("Alice", 85), ("Bob", 92), ("Charlie", 78)]
sorted_students = sorted(students, key=lambda x: x[1], reverse=True)

# map
squared = list(map(lambda x: x**2, numbers))

# filter
evens = list(filter(lambda x: x % 2 == 0, numbers))

# reduce
from functools import reduce
product = reduce(lambda x, y: x * y, numbers)
```

## 3. 装饰器

### 基本装饰器

```python
def timer(func):
    """计时装饰器"""
    import time
    
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} took {end - start:.2f} seconds")
        return result
    
    return wrapper

@timer
def slow_function():
    import time
    time.sleep(1)
    return "Done!"

# 等价于
slow_function = timer(slow_function)
```

### 带参数的装饰器

```python
def repeat(times=3):
    """重复执行指定次数"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            results = []
            for _ in range(times):
                results.append(func(*args, **kwargs))
            return results
        return wrapper
    return decorator

@repeat(times=3)
def say_hello():
    return "Hello!"

print(say_hello())  # ['Hello!', 'Hello!', 'Hello!']
```

### functools.wraps

```python
def my_decorator(func):
    def wrapper(*args, **kwargs):
        """Wrapper function"""
        return func(*args, **kwargs)
    
    wrapper.__name__ = func.__name__
    wrapper.__doc__ = func.__doc__
    wrapper.__module__ = func.__module__
    return wrapper

# 或使用 functools.wraps
from functools import wraps

def my_decorator(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        """Wrapper function"""
        return func(*args, **kwargs)
    return wrapper
```

## 4. 闭包

```python
def make_multiplier(factor):
    """创建乘法函数"""
    def multiply(number):
        return number * factor
    return multiply

# 创建乘以 3 的函数
times_3 = make_multiplier(3)
print(times_3(5))  # 15

# 创建乘以 5 的函数
times_5 = make_multiplier(5)
print(times_5(5))  # 25
```

## 5. 模块与导入

### 模块基础

```python
# 导入整个模块
import math
print(math.pi)
print(math.sqrt(16))

# 导入特定项
from math import pi, sqrt
print(pi)
print(sqrt(16))

# 给模块起别名
import numpy as np
import pandas as pd

# 导入所有项 (不推荐)

```

### 创建模块

```python
# mymodule.py
"""我的模块"""

    return "Hello from module!"

__all__ = ["hello"]  # 控制 from module import * 的行为
```


### collections - 容器类型

```python
from collections import defaultdict, Counter, deque, OrderedDict

# defaultdict - 默认值字典
word_count = defaultdict(int)
for word in ["apple", "banana", "apple", "cherry", "banana", "apple"]:
    word_count[word] += 1
# {'apple': 3, 'banana': 2, 'cherry': 1}

# Counter - 计数器
counter = Counter(["red", "blue", "red", "green", "blue", "blue"])
print(counter.most_common(2))  # [('blue', 3), ('red', 2)]

# deque - 双端队列
dq = deque([1, 2, 3])
dq.appendleft(0)  # [0, 1, 2, 3]
dq.append(4)       # [0, 1, 2, 3, 4]
dq.popleft()       # [1, 2, 3, 4]
```
