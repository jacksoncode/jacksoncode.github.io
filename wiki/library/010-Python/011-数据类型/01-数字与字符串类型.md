# Python 数据类型详解

## 1. 数字类型

### 整数 (int)

```python
# 进制表示
decimal = 42          # 十进制
binary = 0b101010     # 二进制 = 42
octal = 0o52          # 八进制 = 42
hexadecimal = 0x2A    # 十六进制 = 42

# 大整数支持
big_number = 1_000_000_000  # 下划线分隔，更易读

# 整数运算
print(10 // 3)   # 3  - 整除
print(10 ** 3)   # 1000 - 幂运算
print(abs(-42))  # 42 - 绝对值
print(divmod(17, 5))  # (3, 2) - 商和余数
```

### 浮点数 (float)

```python
# 基本运算
x = 3.14
y = 2.0

print(x + y)   # 5.14
print(x * y)   # 6.28
print(x ** 2)  # 9.8596

# 精度问题
a = 0.1 + 0.2
print(a)          # 0.30000000000000004
print(round(a, 2)) # 0.3

# 使用 decimal 模块处理精确计算
from decimal import Decimal
result = Decimal('0.1') + Decimal('0.2')
print(result)  # 0.3
```

### 复数 (complex)

```python
z = 3 + 4j
print(z.real)  # 3.0
print(z.imag)   # 4.0
print(abs(z))   # 5.0 - 模长
```

## 2. 字符串类型

### 字符串创建

```python
# 多种引号方式
s1 = '单引号字符串'
s2 = "双引号字符串"
s3 = '''多行字符串
第一行
第二行'''

# 转义字符
path = "C:\\Users\\Name\\file.txt"
raw_string = r"C:\Users\Name\file.txt"  # 原始字符串

# f-string 格式化
name = "Alice"
age = 25
print(f"{name} is {age} years old")
print(f"{name.upper()}")  # ALICE
print(f"{age:05d}")      # 00025 - 零填充
print(f"{3.14159:.2f}")  # 3.14 - 保留两位小数
```

### 字符串方法

```python
text = "  Hello, Python!  "

# 大小写转换
print(text.upper())           # "  HELLO, PYTHON!  "
print(text.lower())           # "  hello, python!  "
print(text.capitalize())      # "  hello, python!  " (首字母大写)
print(text.title())           # "  Hello, Python!  "

# 去除空白
print(text.strip())           # "Hello, Python!"
print(text.lstrip())         # "Hello, Python!  "
print(text.rstrip())         # "  Hello, Python!"

# 查找与替换
print(text.find("Python"))    # 9 - 索引位置
print(text.replace("Python", "World"))  # "  Hello, World!  "
print("Python" in text)      # True

# 分割与连接
words = "apple,banana,cherry".split(",")
print(words)                  # ['apple', 'banana', 'cherry']
print("-".join(words))       # "apple-banana-cherry"
```

## 3. 列表 (List)

### 列表创建与操作

```python
# 创建列表
empty = []
numbers = [1, 2, 3, 4, 5]
mixed = [1, "hello", 3.14, True]
nested = [[1, 2], [3, 4]]

# 列表推导式
squares = [x**2 for x in range(10)]
even_numbers = [x for x in range(20) if x % 2 == 0]
matrix = [[i*j for j in range(5)] for i in range(5)]
```

### 列表常用操作

```python
fruits = ["apple", "banana", "cherry"]

# 添加元素
fruits.append("date")        # 末尾添加
fruits.insert(0, "avocado")  # 指定位置插入
fruits.extend(["elderberry", "fig"])  # 合并列表

# 删除元素
fruits.remove("banana")       # 移除第一个匹配项
popped = fruits.pop()         # 移除并返回最后一个
del fruits[0]                 # 按索引删除

# 查找
print(fruits.index("cherry")) # 元素索引
print(fruits.count("apple"))   # 元素出现次数
print("apple" in fruits)      # 是否存在

# 排序
fruits.sort()                 # 原地排序
sorted_fruits = sorted(fruits, reverse=True)  # 返回新列表
```

## 4. 元组 (Tuple)

### 元组特性

```python
# 元组是不可变的
point = (10, 20)
# point[0] = 15  # TypeError!

# 单元素元组需要逗号
single = (42,)  # 不是 (42)

# 解包
x, y, z = (1, 2, 3)
first, *rest = [1, 2, 3, 4, 5]  # rest = [2, 3, 4, 5]

# 命名元组
from collections import namedtuple
Point = namedtuple('Point', ['x', 'y'])
p = Point(10, 20)
print(p.x, p.y)  # 10 20
```

## 5. 字典 (Dictionary)

### 字典操作

```python
# 创建字典
person = {"name": "Alice", "age": 30}
person = dict(name="Alice", age=30)  # 关键字参数
pairs = dict([("a", 1), ("b", 2)])   # 从列表转换

# 字典推导式
squares = {x: x**2 for x in range(5)}
filtered = {k: v for k, v in person.items() if v > 25}
```

### 字典方法

```python
user = {"name": "Bob", "age": 25, "city": "Shanghai"}

# 访问
print(user["name"])                    # Bob
print(user.get("country", "Unknown"))  # 安全访问

# 修改
user["age"] = 26                       # 修改值
user["country"] = "China"              # 添加新键值对
user.update({"email": "bob@example.com", "age": 27})  # 批量更新

# 删除
del user["city"]
popped = user.pop("country")          # 移除并返回值

# 遍历
for key in user.keys():
    print(f"{key}: {user[key]}")

for key, value in user.items():
    print(f"{key}: {value}")

for value in user.values():
    print(value)
```

## 6. 集合 (Set)

### 集合操作

```python
# 创建集合
a = {1, 2, 3, 4, 5}
b = {4, 5, 6, 7, 8}
empty = set()  # 注意: {} 是空字典，不是空集合

# 集合运算
print(a | b)   # 并集
print(a & b)   # 交集
print(a - b)   # 差集
print(a ^ b)   # 对称差集 (并集 - 交集)

# 判断关系
print(3 in a)       # True - 成员测试
print(a.issubset(b))    # a 是否是 b 的子集
print(a.issuperset({1, 2}))  # a 是否是 b 的超集
```

## 7. 布尔类型

```python
# 假值 (Falsy)
print(bool(False))     # False
print(bool(None))      # False
print(bool(0))        # False
print(bool(0.0))      # False
print(bool(""))       # False
print(bool([]))       # False
print(bool({}))       # False

# 真值 (Truthy) - 除了以上都是 True
print(bool("hello"))   # True
print(bool([1, 2]))    # True
print(bool(42))        # True
```

## 8. 类型转换

```python
# 字符串转数字
int("42")          # 42
float("3.14")      # 3.14
int("1010", 2)      # 10 - 二进制转十进制
int("FF", 16)       # 255 - 十六进制转十进制

# 数字转字符串
str(42)             # "42"
str(3.14)           # "3.14"
format(255, '02X')  # "FF" - 十六进制格式

# 容器类型转换
list("abc")         # ['a', 'b', 'c']
tuple([1, 2, 3])    # (1, 2, 3)
set([1, 2, 2, 3])   # {1, 2, 3}
dict([["a", 1], ["b", 2]])  # {'a': 1, 'b': 2}
```
